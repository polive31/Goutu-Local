
	/* Custom enqueue stylesheets & scripts */
	public function custom_enqueue_scripts_styles {
		if ( is_singular('recipe') ) {

				wp_enqueue_style( 'fraction', 
					WPUltimateRecipe::get()->coreUrl . '/vendor/fraction-js/index.js', 
					array(), 
					CHILD_THEME_VERSION, 
					true );

			if ( is_admin() ) {
				wp_enqueue_style( 'recipe_admin_stylesheet', 
					WPUltimateRecipe::get()->coreUrl . '/css/admin.css', 
					array(), 
					CHILD_THEME_VERSION, 
					true );
					
			}
			else {
				wp_enqueue_style( 'recipe_frontent_stylesheet', 
					self::$_PluginPath . 'assets/css/custom-recipe.css', 
					array(), 
					CHILD_THEME_VERSION, 
					true );

				wp_register_style( 'print_button', 
					WPUltimateRecipe::get()->coreUrl . '/js/print_button.js', 
					array('jquery'), 
					CHILD_THEME_VERSION, 
					true );
					
				wp_localize_script( 'print_button',
					'data',
					array(
            'name' => 'wpurp_print',
            'ajaxurl' => WPUltimateRecipe::get()->helper('ajax')->url(),
            'nonce' => wp_create_nonce( 'wpurp_print' ),
            'custom_print_css_url' => get_stylesheet_directory_uri() . '/assets/css/custom-recipe-print.css',
            'coreUrl' => WPUltimateRecipe::get()->coreUrl,
            'premiumUrl' => WPUltimateRecipe::is_premium_active() ? WPUltimateRecipePremium::get()->premiumUrl : false,
            'title' => __('Print this Recipe','foodiepro'),
            'permalinks' => get_option('permalink_structure'),
	         ),
	         )


			}

	}
